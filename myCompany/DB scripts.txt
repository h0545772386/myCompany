	[WH150P] [decimal] (6, 2) NOT NULL DEFAULT 0,
	[WH175P] [decimal] (6, 2) NOT NULL DEFAULT 0,

	[WrkrNumber] [int] NOT NULL DEFAULT 0,
	[IsHourly] [bit] NOT NULL DEFAULT 0,
	[HourlyPrice] [decimal] (6, 2) NOT NULL DEFAULT 0,
